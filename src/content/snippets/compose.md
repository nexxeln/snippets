// deno-lint-ignore ban-types
export function compose(fn1: Function, ...fns: Function[]) {
  if (fns.length === 0) {
    return fn1;
  }

  return (...args: unknown[]) =>
    fns.reduce((prev, fn) => fn(prev), fn1(...args));
}
